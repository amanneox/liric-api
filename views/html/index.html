$ch = curl_init(); 

$url='https://liric.herokuapp.com/api/song/Brand%20New'; 

curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false); 


curl_setopt($ch, CURLOPT_RETURNTRANSFER, true); 


curl_setopt($ch, CURLOPT_URL,$url); 


$result=curl_exec($ch); 


curl_close($ch); 

var_dump(json_decode($result, true));
